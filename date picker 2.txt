import tkinter as tk
from tkcalendar import DateEntry

# Variable to store selected date
selected_date = None


def on_date_selected(event):
    global selected_date
    selected_date = date_entry.get()
    print("Date selected and stored:", selected_date)


def on_date_selected1(event):
    global selected_date
    selected_date2 = date_entry1.get()
    print("Date selected and stored:", selected_date2)


# Set up main window

root = tk.Tk()
root.title("Auto-Store Date from Calendar")
root.minsize(350, 250)  # Set minimum window size
tk.Label(root, text="Choose a start Date", background='gray61', foreground="white").pack(padx=20, pady=20)

# Create DateEntry widget
date_entry = DateEntry(root)
date_entry.pack(pady=10)

tk.Label(root, text="Choose a End Date", background='gray61', foreground="white").pack(padx=20, pady=20)
# Create Second DateEntry widget
date_entry1 = DateEntry(root)
date_entry1.pack(pady=10)

# Bind the event that fires when a date is selected
date_entry.bind("<<DateEntrySelected>>", on_date_selected)
date_entry1.bind("<<DateEntrySelected>>", on_date_selected1)

root.mainloop()
